            var options = {
                // 가능한 경우, 높은 정확도의 위치(예를 들어, GPS 등) 를 읽어오려면 true로 설정

        //좌표값 세팅
        var setGpsFunc = function(initSet) {
            var gpsx = $('#gpsx').val();
            var gpsy = $('#gpsy').val();
            var coordinateRadius = $('select[name=coordinateRadius]').val();

            //class 삭제
            for(var i=1; i<12; i++){
                var coorClass = $('#coorClass' + i);
                coorClass.removeClass("on");
            }

            $('#coorClass1').addClass("on");
            $('#coorClass6').addClass("on");

            // 주소명 넣기
            // 주소-좌표 변환 객체를 생성합니다
            var geocoder = new daum.maps.services.Geocoder();

            // 좌표로 행정동 주소 정보를 요청합니다
            geocoder.coord2RegionCode(gpsy, gpsx, function(result, status) {
                if(status === daum.maps.services.Status.OK){
                    var txtLoc = $('#txt_loc');
                    txtLoc.text(result[0].region_1depth_name + ' ' + result[0].region_2depth_name);
                }
            });

            //총건수
            setTotalCnt(gpsx, gpsy, coordinateRadius);

            //공연/전시
            setExperience(initSet, gpsx, gpsy, coordinateRadius, '06');

            //공연장
            setFacility(initSet, gpsx, gpsy, coordinateRadius, 100);
        }

        //공연/전시 type 06 / 행사축제 type 50
        var setExperience = function(initSet, gpsx, gpsy, coordinateRadius, genreType ) {
            $.ajax({
                url			:	'/setExperience.ajax'
                ,type		:	'get'
                ,data		:	{
                    'gpsx' : gpsx
                    , 'gpsy' : gpsy
                    , 'coordinateRadius' : coordinateRadius
                }
                ,dataType	:	'html'
                ,success	:	function( res ) {
                    var $areaSlider1 = $('.area_slider1');
                    var liTag = '';

                    var experienceObj = JSON.parse(res);

                    //제한걸기
                    var limitVal = 0;

                    for(var i=0; i<experienceObj.length; i++){
                        if(experienceObj[i].TYPE == genreType){

                            if(limitVal > 11){
                                break;
                            }

                            var uci = experienceObj[i].UCI;
                            uci = uci.replace("+", "%2B");


                            var imgUrl = '/assets/images/main/noimage.jpg';
                            if(experienceObj[i].REFERENCE_IDENTIFIER_ORG != undefined && experienceObj[i].REFERENCE_IDENTIFIER_ORG != null){
                                imgUrl = '/upload/rdf/' + experienceObj[i].REFERENCE_IDENTIFIER_ORG;
                            }

                            if(experienceObj[i].REFERENCE_IDENTIFIER != undefined && experienceObj[i].REFERENCE_IDENTIFIER != null){
                                imgUrl = '/upload/rdf/' + experienceObj[i].REFERENCE_IDENTIFIER;
                            }

                            //화면 문제로 인한 ... 처리
                            var hallName = experienceObj[i].NAME;
                            if(hallName.length > 15){
                                hallName = hallName.substr(0, 15) + '...';
                            }

                            liTag = liTag +
                                '<li>' +
                                '<a href="/perform/performView.do?uci='+ uci +'">' +
                                '<div class="thumb">' +
                                // '<img src="/upload/rdf/' + experienceObj[i].REFERENCE_IDENTIFIER_ORG + '" alt="' + experienceObj[i].TITLE + '">' +
                                '<img src="' + imgUrl + '" alt="' + experienceObj[i].TITLE + '">' +
                                '<span class="label">' + experienceObj[i].TITLE +
                                '</span>' +
                                '</div>' +
                                '<dl>' +
                                '<dt>' + experienceObj[i].TITLE + '</dt>' +
                                '<dd>' +
                                '<span>' + experienceObj[i].SIDO + '</span>' +
                                '<span>' + hallName + '</span>' +
                                '<span class="distance">' + experienceObj[i].coorDist + 'km</span>' +
                                '</dd>' +
                                '</dl>' +
                                '</a>' +
                                '</li>'
                            ;

                            limitVal = limitVal + 1;
                        }
                    }

                    if(limitVal == 0){
                        liTag = liTag + '<li><div class="thumb"></div><dl><dt></dt><dd></dd></dl></li>';
                    }

                    //최초가 아닌경우
                    if(initSet == 1){
                        //slick 삭제
                        $areaSlider1.slick('unslick');
                        $areaSlider1.remove();

                        //삭제한 tag 다시 넣기
                        var $inner1 = $('#inner1');
                        var ulTag = '<ul class="area_slider1">' + '</ul>';
                        $inner1.html(ulTag);

                        var $reAreaSlider1 = $('.area_slider1');

                        $reAreaSlider1.html(liTag);

                        setAreaSlider($reAreaSlider1, 1);
                    }else{
                        $areaSlider1.html(liTag);

                        setAreaSlider($areaSlider1, 1);
                    }
                }
                ,error		:	function(data, status, err) {
                }
            });
        }

                speed: 500,

                slidesToScroll: 1,

                    {
                        breakpoint: 767,
                        settings: {
                            slidesToShow: 2,
                            slidesToScroll: 1
                        }
                    }
                ]
            });

            if(type == 1){
                $('.area_ctrl1 .btn_prev').click(function() {
                    $areaSlider.slick('slickPrev');
                });
                $('.area_ctrl1 .btn_next').click(function() {
                    $areaSlider.slick('slickNext');
                });
            }else{
                $('.area_ctrl2 .btn_prev').click(function() {
                    $areaSlider.slick('slickPrev');

                    $areaSlider.slick('slickNext');

        var selectRegion = function(pname, name) {


            var regionName = pname + ' ' + name;

            var txtLoc = $('#txt_loc');
            txtLoc.text(regionName);

            // 주소-좌표 변환 객체를 생성합니다
            var geocoder = new daum.maps.services.Geocoder();

            geocoder.addressSearch(regionName, function(result, status) {
                if (status === daum.maps.services.Status.OK) {
                    $('#gpsx').val(result[0].y);
                    $('#gpsy').val(result[0].x);

                    setGpsFunc(1);
                }else{
                    $('#gpsx').val(37.576073);
                    $('#gpsy').val(126.976923);

                    setGpsFunc(1);
                }
            });
        }
